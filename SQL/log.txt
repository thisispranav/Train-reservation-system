CREATE DATABASE TRS;

USE TRS;

CREATE TABLE INFO(
ID BIGINT,
NAME CHAR(25),
GEN VARCHAR(10),
AGE VARCHAR(10),
MOBNO BIGINT,
EMAIL VARCHAR(100),
PASSWORD VARCHAR(100),
PRIMARY KEY(ID,MOBNO));

CREATE TABLE ADMIN(
ADMIN CHAR(25),
MOBNO BIGINT,
PASSWORD VARCHAR(100));

CREATE TABLE TRAIN(
TNO BIGINT,
TNAME VARCHAR(50),
TROUTE VARCHAR(30),
TIME VARCHAR(10),
TIME2 VARCHAR(10),
PRIMARY KEY (TNAME));

CREATE TABLE CTOB(
STOPS VARCHAR(50),
EORP VARCHAR(30));

CREATE TABLE CTOT(
STOPS VARCHAR(50),
EORP VARCHAR(30));

CREATE TABLE TICKETS(
ID BIGINT,
TNO BIGINT,
TNAME VARCHAR(50),
DATE VARCHAR(10),
SEATS INT,
SEATNO VARCHAR(50),
START VARCHAR(50),
END VARCHAR(50),
TIME VARCHAR(20),
MN CHAR(1),
TICKNO BIGINT,
PRIMARY KEY (TICKNO)
);

create table date(date date);
